Работа с файлами и каталогами:
1. pwd - путь к текущему каталогу
2. cd - домашний каталог текущего пользователя 
3. cd / - корневой каталог
4. cd ../.. - на 2 уровня вперед 
5. cd - - прошлая папка
6. ls - содержимое текущей папки 
7. ls -l - содержимое текущей папки с их подробной информацией 
8. ls -a - показывает скрытые файлы и папки
9. mkdir new1 - создание нового каталога
10. mkdir new1 new2 - создание несколько каталогов
11. mkdir new1/new2 - создание дерева каталога
12. rm "название файла" - удаление файла 
13. rm -r "название каталога" - удаление каталога
14. mv "тек. назв файла" "новое назв файла" - переим
15. mv "название файла" "название каталога" - перемещение файла в новый каталог
16. mv "название каталога" "название каталога" - перемещение каталога в другой каталог
17. touch "название файла" - создание файла 
18. cat "название файла" - открытие файла в терминале
